let component = new RGUI.Component({
    template: template,
    data: {
        flavors: []
    },
    init() {
        setTimeout(() => {
            this.data.flavors = [
                {memory: 1, price: 0.23},
                {memory: 2, price: 0.52},
                {memory: 4, price: 0.93},
                {memory: 8, price: 1.86},
                {memory: 16, price: 3.60},
            ];
            this.data.memoryPrice = this.data.flavors[0].price;
            this.$update();
        }, 500);
    }
});
